Node.js讲义

cookie在登陆中的应用

// 1. 引入模块
var http = require('http')
var querystring = require('querystring')
var fs = require('fs')
var path = require('path')


// 2. 创建服务器对象
var server = http.createServer()

// 3. 开启服务并监听端口
server.listen(3000,()=>{
  console.log('the server is running at http://127.0.0.1:3000');
})

// 4.注册请求事件,监听请求
server.on('request',(req,res)=>{
  bindRender(req,res); // 绑定两个对象
  var url = req.url;
  console.log(req.headers.cookie);
  var method = req.method;
  var cookie = querystring.parse(req.headers.cookie,'; ','=',true)
  // console.log(cookie);
  if(method=='GET'&&url=='/'){
    if (cookie.isFlag == 'true') {
      res.render('index')
    }else {
      res.writeHeader(302,{
        "location":'/login'
      })
      res.end()
    }

  }else if(url=='/index'&&method=='GET'){
    if (cookie.isFlag == 'true') {
      res.render('index')
    } else {  
      res.writeHeader(302, {
        "location": '/login'
      })
      res.end()
    }
  } else if (method == 'GET' && url == '/login') {
    res.render('login')
  }else if(method=='POST'&&url=='/login'){
    var dataStr = ''
    req.on('data',(chunk)=>{
      dataStr += chunk;
    })
    req.on('end',()=>{
      var dataObj = querystring.parse(dataStr);
      // console.log(dataObj);
      if(dataObj.userName=='tom'&&dataObj.userPwd=='123'){

        res.writeHeader(302,{
          'Set-Cookie': ['isFlag=true'],
          'location':'/index' // 跳转到主页面
        })
        res.end();
      }
    })
  }else {
    res.end('404')
  }
})


function bindRender(req,res){
  res.render = function (fileName) {
    fs.readFile(path.join(__dirname, './views/' + fileName + '.html'), (err, data) => {
      if (err) return console.log(err.message);
      res.end(data);
    })
  }
}
