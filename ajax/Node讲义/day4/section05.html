EJS模板引擎渲染页面

ejs里面渲染数据内容的方式与art-template的原生语法是类似的,也是使用<%= %>的方式来输出内容,<%%>内部可以写一切表达式,判断,循环,三元...都可以。

index.ejs中的内容如下:

<!DOCTYPE html>
<html lang="en">

<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <meta http-equiv="X-UA-Compatible" content="ie=edge">
  <title>Document</title>
</head>

<body>
  <h2>这是一个主页面的内容</h2>
  <p>姓名:<%= name%></p>
  <p>性别:<%= gender%></p>
  <p>年龄:<%= age%></p>
  <p>爱好:</p>
  <ul>
    <% hobby.forEach(item=>{ %>
      <li><%= item%></li>
    <%})%>
  </ul>
</body>

</html>

app.js中的内容如下:

// 1. 引入模块
var express = require('express')
var ejs = require('ejs')

// 2. 创建服务器对象
var app = express();

// 3. 启动服务,并监听端口
app.listen(3000,()=>{
  console.log('express server is running at http://127.0.0.1:3000');
})

// 4. 注册中间件
// 设置一个指定的模板引擎ejs，其中第一个参数的值是固定写法.
app.set('view engine','ejs');
// 第二个app.set是指定要渲染的页面的来源,第一个参数也是一个固定的写法,表示是要渲染的是静态页面
app.set('views','./views')
app.get('/', (req, res) => {
  // res.send('这是主页面的内容...')
  res.render('index',{name:'tom',age:20,gender:'男',hobby:['吃饭','睡觉','打豆豆']})
})
app.get('/movie', (req, res) => {
  // res.send('这是一个电影页面的内容啊...')
  res.render('movie',{})
})
app.get('/login', (req, res) => {
  // res.send('这是一个用户登陆的请求啊......')
  res.render('login',{})
})

app.post('/',(req, res) => {
  res.send('这是一个post请求啊...')
})
